0026 function mredge_reorganize_by_frequency(series_number, subdir, driving_frequencies)
0027   NIF_EXT = '.nii.gz';
0028   for f = 1:numel(driving_frequencies)
0029       index = (f-1)*3;
0030       for c = 1:3
0031           series_path = fullfile(subdir, num2str(series_number), num2str(index+c));
0032           frequency_path = fullfile(subdir, num2str(driving_frequencies(f)), num2str(c));
0033           if ~exist(frequency_path, 'dir')
0034               mkdir(frequency_path);
0035           end
0036           movefile(fullfile(series_path, mredge_filename(series_number,  num2str(index+c), NIF_EXT)), fullfile(frequency_path, mredge_filename(driving_frequencies(f),  c, NIFTI_EXTENSION)));
0037       end
0038   end
0039   series_path = fullfile(subdir, num2str(series_number));
0040   if exist(series_path, 'dir')
0041       rmdir(series_path, 's');
0042   end
